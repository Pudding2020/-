# 直接插入排序

o(n^2)  o(1)  稳定

'''
void swap(vector<int> &nums,int i,int j)
{
    int tmp=nums[i];
    nums[i]=nums[j];
    nums[j]=tmp;
}
void sort(vector<int> &nums,int left,int right)
{
    for(int i=1;i<=right;++i)
      for(int j=i;j>0 && nums[j-1]>nums[j];--j)
        swap(nums,j-1,j);
}
